```lua
-- Enable Notifcation Sound
Config.EnableNotifcationSound = true

-- Set Notifcation Sound Volume (1.0 = 100% / 0.0 = 0%)
Config.NotifcationSoundVolume = 1.0

-- Enable Promise Sound
Config.EnablePromiseSound = true

-- Set Promise Sound Volume (1.0 = 100% / 0.0 = 0%)
Config.PromiseSoundVolume = 1.0

-- top-left | top-right | top-center | bottom-left | bottom-right | bottom-center | right-center | left-center
Config.DefaultPosition = 'right-center'

-- bounce | slide | zoom | flip
Config.Transition = 'bounce'

-- light | dark | colored | custom
Config.Theme = 'dark'

-- Hide ProgressBar
Config.HideProgressBar = false

-- Notification Limits
Config.NotifcationLimit = 5
```
